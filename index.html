High-performance system utilities by Rushikesh Sakharle

Available Packages

  • global-logrotate v2.1.15 - Fast parallel log rotation with AES-256 encryption

APT Repository (Debian/Ubuntu)

# Add repository
echo "deb [trusted=yes] https://rushikeshsakharleofficial.github.io/global-sys-utils/apt stable main" | sudo tee /etc/apt/sources.list.d/global-sys-utils.list

# Install
sudo apt update
sudo apt install global-logrotate

YUM Repository (RHEL/CentOS/Fedora)

# Add repository
sudo tee /etc/yum.repos.d/global-sys-utils.repo << 'REPO'
[global-sys-utils]
name=Global Sys Utils
baseurl=https://rushikeshsakharleofficial.github.io/global-sys-utils/rpm/$basearch/
enabled=1
gpgcheck=0
REPO

# Install
sudo yum install global-logrotate
